CREATE MULTISET TABLE Customer (
    CustomerID INTEGER NOT NULL,
    FirstName VARCHAR(50) CHARACTER SET UNICODE NOT NULL,
    LastName VARCHAR(50) CHARACTER SET UNICODE NOT NULL,
    Email VARCHAR(100) CHARACTER SET UNICODE,
    PhoneNumber VARCHAR(15) CHARACTER SET UNICODE,
    BirthDate DATE,
    RegistrationDate TIMESTAMP(6),
    PRIMARY KEY (CustomerID)
);
